"About loops.";

$def! l list "a" "b" "c" "d";

info "while loops: test a list element in a list";
$defl! list-contains? (l x)
(
	$def! not-found #t;
	$while ($and not-found (not? (null? l)))
	(
		$if (eqv? (first l) x)
		(
			$def! not-found #f;
		)
		(
			$def! l rest l;
		)
	);
	not? not-found
);
$check list-contains? l "b";
$check-not list-contains? l "e";
